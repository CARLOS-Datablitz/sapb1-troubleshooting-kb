SELECT
    MBT.BLOCKED_CONNECTION_ID          AS CONEXION_BLOQUEADA,
    MBT.LOCK_OWNER_CONNECTION_ID       AS CONEXION_BLOQUEADORA,
    CB.USER_NAME                       AS USUARIO_BLOQUEADO,
    CB.CLIENT_HOST                     AS PC_BLOQUEADA,
    CO.USER_NAME                       AS USUARIO_BLOQUEADOR,
    CO.CLIENT_HOST                     AS PC_BLOQUEADORA,
    MBT.WAITING_SCHEMA_NAME,
    MBT.WAITING_TABLE_NAME,
    MBT.WAITING_OBJECT_NAME,
    MBT.LOCK_TYPE,
    MBT.LOCK_MODE,
    MBT.BLOCKED_TIME
FROM SYS.M_BLOCKED_TRANSACTIONS MBT
LEFT JOIN SYS.M_CONNECTIONS CB
       ON MBT.BLOCKED_CONNECTION_ID = CB.CONNECTION_ID
LEFT JOIN SYS.M_CONNECTIONS CO
       ON MBT.LOCK_OWNER_CONNECTION_ID = CO.CONNECTION_ID
ORDER BY MBT.BLOCKED_TIME DESC;

 ============ Ideal para detectar sesiones “colgadas” =================
 
 SELECT
    CONNECTION_ID,
    USER_NAME,
    CLIENT_HOST,
    START_TIME
FROM SYS.M_CONNECTIONS
WHERE USER_NAME NOT LIKE '%SYS%'
ORDER BY START_TIME;

 ============ Ideal para detectar sesiones “bloquedas” =================
SELECT
    BLOCKED_CONNECTION_ID,
    LOCK_OWNER_CONNECTION_ID,
    WAITING_SCHEMA_NAME,
    WAITING_TABLE_NAME,
    LOCK_TYPE,
    LOCK_MODE,
    BLOCKED_TIME
FROM SYS.M_BLOCKED_TRANSACTIONS;
